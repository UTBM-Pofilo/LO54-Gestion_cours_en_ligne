DROP TABLE CLIENT;
DROP TABLE COURSE_SESSION;
DROP TABLE LOCATION;
DROP TABLE COURSE;

CREATE TABLE LOCATION
    (ID INT PRIMARY KEY,
    CITY VARCHAR(8) NOT NULL);
	
	
CREATE TABLE COURSE
	(CODE VARCHAR(8) PRIMARY KEY,
	TITLE VARCHAR(20) NOT NULL);
	
	
CREATE TABLE COURSE_SESSION
	(ID INT PRIMARY KEY,
	START_DATE DATE NOT NULL,
	END_DATE DATE NOT NULL,
	COURSE_CODE VARCHAR(8),
	LOCATION_ID INT NOT NULL);
	
ALTER TABLE COURSE_SESSION ADD FOREIGN KEY (COURSE_CODE) REFERENCES COURSE (CODE);
ALTER TABLE COURSE_SESSION ADD FOREIGN KEY (LOCATION_ID) REFERENCES LOCATION (ID);

CREATE TABLE CLIENT
	(ID INT PRIMARY KEY,
	LAST_NAME VARCHAR(14) NOT NULL,
	FIRST_NAME VARCHAR(14) NOT NULL,
	ADDRESS VARCHAR(14) NOT NULL,
	PHONE VARCHAR(10) NOT NULL,
	EMAIL VARCHAR(14),
	COURSE_SESSION_ID INT);
	
ALTER TABLE CLIENT ADD FOREIGN KEY (COURSE_SESSION_ID) REFERENCES COURSE_SESSION (ID);
